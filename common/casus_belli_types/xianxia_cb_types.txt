xianxia_seize_spirit_vein_cb = {
    icon = de_jure_cb
    group = de_jure
    war_name = XIANXIA_SEIZE_SPIRIT_VEIN_WAR_NAME
    war_name_base = XIANXIA_SEIZE_SPIRIT_VEIN_WAR_NAME_BASE
    allowed_for_character = { scope:attacker = { is_landed = yes is_cultivator = yes } }
    allowed_against_character = { scope:defender = { is_landed = yes } }
    ai_only_against_neighbors = yes
    should_invalidate = { scope:defender = { is_landed = no } }
    on_invalidated_desc = xianxia_cb_invalidated_desc
    on_victory_desc = {
        first_valid = {
            triggered_desc = { trigger = { scope:attacker = { is_local_player = yes } } desc = xianxia_seize_spirit_vein_victory_desc_attacker }
            triggered_desc = { trigger = { scope:defender = { is_local_player = yes } } desc = xianxia_seize_spirit_vein_victory_desc_defender }
            desc = xianxia_seize_spirit_vein_victory_desc
        }
    }
    on_white_peace_desc = {
        first_valid = {
            triggered_desc = { trigger = { scope:defender = { is_local_player = yes } } desc = xianxia_seize_spirit_vein_white_peace_desc_defender }
            desc = xianxia_seize_spirit_vein_white_peace_desc
        }
    }
    on_defeat_desc = {
        first_valid = {
            triggered_desc = { trigger = { scope:attacker = { is_local_player = yes } } desc = xianxia_seize_spirit_vein_defeat_desc_attacker }
            desc = xianxia_seize_spirit_vein_defeat_desc
        }
    }
    cost = {
        prestige = {
            value = 0
            add = 250
            multiply = common_cb_prestige_cost_multiplier
        }
    }
    on_victory = {
        scope:attacker = { add_prestige = 180 change_variable = { name = sect_authority_score add = 6 } add_character_modifier = { modifier = sect_resource_stockpile days = 1825 } }
        scope:defender = { add_character_modifier = { modifier = sect_resource_obligations days = 1825 } }
        scope:defender.capital_county = { add_county_modifier = { modifier = spirit_vein_contested days = 1825 } }
        add_truce_attacker_victory_effect = yes
    }
    on_defeat = { scope:attacker = { add_prestige = -120 } add_truce_attacker_defeat_effect = yes }
    on_white_peace = { add_truce_white_peace_effect = yes }
    # ai_will_do removed for CK3 1.19 casus belli parser compatibility

}
