# Jianghu Dynasty v2.10.0 - lightweight vanilla-map scripted triggers.

jdtc_adult_free_character_trigger = {
	is_adult = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
}

jdtc_martial_capable_character_trigger = {
	jdtc_adult_free_character_trigger = yes
	OR = {
		prowess >= 8
		martial >= 8
	}
}

jdtc_court_capable_character_trigger = {
	jdtc_adult_free_character_trigger = yes
	OR = {
		diplomacy >= 8
		stewardship >= 8
		learning >= 8
	}
}
